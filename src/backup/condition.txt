import React, { useState } from 'react';
import Component1 from './Component1';
import Component2 from './Component2';
import Component3 from './Component3';
import Component4 from './Component4';
import Component5 from './Component5';

function MyComponent() {
  const [componentToShow, setComponentToShow] = useState(null);

  const handleShowComponent1 = () => {
    setComponentToShow('Component1');
  }

  const handleShowComponent2 = () => {
    setComponentToShow('Component2');
  }

  const handleShowComponent3 = () => {
    setComponentToShow('Component3');
  }

  const handleShowComponent4 = () => {
    setComponentToShow('Component4');
  }

  const handleShowComponent5 = () => {
    setComponentToShow('Component5');
  }

  return (
    <div>
      <button onClick={handleShowComponent1}>
        Show Component 1
      </button>
      <button onClick={handleShowComponent2}>
        Show Component 2
      </button>
      <button onClick={handleShowComponent3}>
        Show Component 3
      </button>
      <button onClick={handleShowComponent4}>
        Show Component 4
      </button>
      <button onClick={handleShowComponent5}>
        Show Component 5
      </button>
      {componentToShow === 'Component1' && <Component1 />}
      {componentToShow === 'Component2' && <Component2 />}
      {componentToShow === 'Component3' && <Component3 />}
      {componentToShow === 'Component4' && <Component4 />}
      {componentToShow === 'Component5' && <Component5 />}
    </div>
  );
}

export default MyComponent;
